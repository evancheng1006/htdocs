#include <stdio.h>

int main() {
    int res = 0; //記錄答案
    int sign = -1; //表示上一個數字的正負號 1表示正數 -1表示負數
    int length = 0; //表示到目前為止找到的 「正負正負」序列有多長
    int x;
    while (scanf("%d", &x)) {
        if (x == 0) break; //結束條件

        //計算序列長度
        if (x > 0) {
            if (sign == -1)
                length += 1;
            else
                length = 1;
            sign = 1;
        }
        else if (x < 0) {
            if (sign == 1)
                length += 1;
            else
                length = 0;
            sign = -1;
        }

        //更新答案
        if (length > res && length > 1 && length%2==1) {
            res = length;
        }
    }
    printf("%d\n", res);
    return 0;
}

用下表解釋 length 和 count 的運作方式：(搭配看Code會更清楚)

  1. 讀到 1，修改正負號為sign=1，length清空，count=1表示有一個正整數。
  2. 讀到 1，正負號一樣，count++，因為count<k，不處理。
  3. 讀到 1，正負號一樣，count++，因為count=k，length++，count清空，sign變號。
  4. 讀到 -1，正負號一樣(剛剛有變號)，count++...
  5. ...依此類推
k = 3 的狀況

 sign     :  +  +  -  -  -  +  +  +  -

完整Code

#include <stdio.h>

int main() {
    int k;
    scanf("%d", &k);

    int res = 0; //記錄答案
    int sign = -1; //表示上一個數字的正負號 1表示正數 -1表示負數
    int length = 0; //表示到目前為止找到的 「正負正負」序列有多長
    int count = 0;
    int x;
    while (scanf("%d", &x)) {
        if (x == 0) break; //結束條件

        //計算序列長度
        if (x > 0) {
            if (sign == 1){
                count += 1;
            }
            else {
                if ( count==0 && length>0 ) { //還沒有讀到負數過
                    count = 0;
                    length = 1;
                } else {          //清空
                    sign = 1;
                    count = 1;
                    length = 0;
                }
            }
            if (count == k) {
                count = 0;
                sign = -1;
                length += 1;
            }
        }
        else if (x < 0) {
            if (sign == -1 && length>0) { //確認前面有讀到正數過
                count += 1;
            }
            else { //清空
                sign = -1;
                count = 0;
                length = 0;
            }
            if (count == k) {
                count = 0;
                sign = 1;
                length += 1;
            }
        }

        //更新答案
        if (length > res && length > 1 && length%2==1) {
            res = length;
        }
    }
    printf("%d\n", res*k);
    return 0;
}
